\poemtitle{A Shropshire Lad LVII}
\begin{verse}
I promise nothing: friends will part; \\
\hspace{0.5cm} All things may end, for all began; \\
And truth and singleness of heart \\
\hspace{0.5cm} Are mortal even as is man.
 
But this unlucky love should last \\
\hspace{0.5cm} When answered passions thin to air; \\ 
Eternal fate so deep has cast \\
\hspace{0.5cm} Its sure foundation of despair.
\end{verse}


\poemtitle{More Poems XII}
\begin{verse}
You smile upon your friend to-day, \\
\hspace{0.5cm} To-day his ills are over; \\
You hearken to the lover's say, \\
\hspace{0.5cm} And happy is the lover.

'Tis late to hearken, late to smile, \\
\hspace{0.5cm} But better late than never; \\
I shall have lived a little while \\
\hspace{0.5cm} Before I die for ever.
\end{verse}
\attrib{A.E. Housman (1859--1936)}
